import java.util.Scanner;

class Buffer{
	int item, num;
	boolean avai=false;
	
	Buffer(int x){
		num=x;
	}
	public synchronized void produce(int value) throws InterruptedException{
		try {
			while(avai) {
				wait();
			}
			item=value;
			avai=true;
			System.out.println("Producer "+num+" put: "+item);
			notify();
		}catch(InterruptedException e) {
			System.out.println(e);
		}
	}
	public synchronized void consume(int value) throws InterruptedException{
		try {
			while(!avai) {
				wait();
			}
			item=value;
			avai=false;
			System.out.println("Consumer "+num+" got: "+item);
			Thread.yield();
			notify();
		}catch(InterruptedException e) {
			System.out.println(e);
		}
	}
}
class Producer extends Thread{
	Buffer buff;
	Producer(Buffer b){
		buff=b;
	}
	public void run() {
		try {
			for(int i=1;i<=5;i++) {
				buff.produce(i);
				Thread.sleep(1000);
			}
		}catch(InterruptedException e) {
			System.out.println(e);
		}
	}
}

class Consumer extends Thread{
	Buffer buff;
	Consumer(Buffer b){
		buff=b;
	}
	public void run() {
		try {
			for(int i=1;i<=5;i++) {
				buff.consume(i);
				Thread.sleep(1000);
			}
		}catch(InterruptedException e) {
			System.out.println(e);
		}
	}
}
public class Prgm8 {

	public static void main(String[] args) {

		Scanner sc= new Scanner(System.in);
		
		System.out.println("Enter number: ");
		int num= sc.nextInt();
		
		Buffer b=new Buffer(num);
		Producer p=new Producer(b);
		Consumer c=new Consumer(b);
		p.start();
		c.start();
	}

}
