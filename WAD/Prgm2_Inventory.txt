<script>
function addItem(){
let item = document.getElementById('item').value;
let quan = document.getElementById('quan').value;
let price = document.getElementById('price').value;

if( !item|| !quan|| !price ) return;

let table= document.getElementById('table');
let amount=quan*price;
let row= table.insertRow()
row.innerHTML = `
        <td>${item}</td>
        <td>${quan}</td>
        <td>${price}</td>
        <td>${amount}</td>
        <td>  <button class="btn btn-sm btn-primary" onclick="editRow(this)">Edit</button>
         <button class="btn bg-dark text-white"
                onclick="this.parentElement.parentElement.remove()">
                Remove
            </button></td>`;


document.getElementById('item').value='';
document.getElementById('quan').value='';
document.getElementById('price').value='';
document.getElementById('amount').value='';
}
</script>
<script>
function editRow(btn) {
  let row = btn.closest("tr");

  let item = prompt("Update item name:", row.cells[0].innerText);
  let quan = prompt("Update quantity:", row.cells[1].innerText);
  let price = prompt("Update price:", row.cells[2].innerText);

  if (item && quan && price) {
    row.cells[0].innerText = item;
    row.cells[1].innerText = quan;
    row.cells[2].innerText = price;
  }
}
</script>
